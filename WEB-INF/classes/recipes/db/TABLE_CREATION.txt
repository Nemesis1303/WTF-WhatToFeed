CREATE TABLE RECIPES ( RECIPE_ID INT UNSIGNED NOT NULL AUTO_INCREMENT,
					   RECIPE_NAME VARCHAR(32) NOT NULL,
					   AUTHOR VARCHAR(16) NOT NULL,
					   RECIPE_ORIGIN_COUNTRY VARCHAR(16),
					   RECIPE_STYLE VARCHAR (32) NOT NULL,
					   RECIPE_TYPE VARCHAR (32) NOT NULL,
					   RECIPE_ELABORATION_STEPS LONGTEXT NOT NULL,
					   PRIMARY KEY (RECIPE_ID));
					  
CREATE TABLE INGREDIENTS( INGREDIENT_ID INT UNSIGNED NOT NULL AUTO_INCREMENT,
						   INGREDIENT_NAME VARCHAR(32) NOT NULL,
						   INGREDIENT_QUANTITY FLOAT NOT NULL,
						   INGREDIENT_QUNATITY_UNITS VARCHAR(16) NOT NULL,
						   PRIMARY KEY (INGREDIENT_ID)); 
						   
CREATE TABLE INGREDIENTS_RECIPES (RECIPE_ID INT UNSIGNED NOT NULL,
								  INGREDIENT_ID INT UNSIGNED NOT NULL,
								  PRIMARY KEY (RECIPE_ID, INGREDIENT_ID),
								  FOREIGN KEY (RECIPE_ID) REFERENCES RECIPES(RECIPE_ID),
								  FOREIGN KEY (INGREDIENT_ID) REFERENCES INGREDIENTS(INGREDIENT_ID));
